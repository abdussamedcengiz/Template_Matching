package com.cengiz.templatematching;

import org.opencv.core.Core;
import org.opencv.core.Core.MinMaxLocResult;
import org.opencv.core.Mat;
import org.opencv.core.Point;
import org.opencv.core.Scalar;
import org.opencv.imgcodecs.Imgcodecs;
import org.opencv.imgproc.Imgproc;

public class Main {
    public static void main(String[] args) {
        System.loadLibrary(Core.NATIVE_LIBRARY_NAME);
        
        Mat source = null;
        Mat template = null;
        String filepath = "C:\\Users\\SAMET\\Downloads\\drive-download-20230525T092359Z-001.zip\\";
        
        source = Imgcodecs.imread(filepath + "31.png");
        template = Imgcodecs.imread(filepath + "template.png");
        
        Mat outputImage = new Mat();
        int matchMethod = Imgproc.TM_CCOEFF;
        Imgproc.matchTemplate(source, template, outputImage, matchMethod);
        
        MinMaxLocResult mmr = Core.minMaxLoc(outputImage);
        Point matchLoc = mmr.maxLoc;
        Imgproc.rectangle(source, matchLoc, new Point(matchLoc.x + template.cols(), matchLoc.y + template.rows()), new Scalar(255, 255, 255));
        
        Imgcodecs.imwrite(filepath + "sonuc.jpg", source);
        System.out.println("Görüntü tarama işlemi tamamlandı.");
    }
}
